ejemplo de tabla de estudiantes:

<div class="container">
  <div class="row">
    <h3>LISTADO DE ESTUDIANTES</h3>
  </div>
  <br>
  <table class="table" *ngIf="listEstudiantes.length > 0">
    <thead>
      <th>Nro</th>
      <th>Nombre</th>
      <th>Estados</th>
    </thead>
    <tbody>
      <tr *ngFor= "let estudiante of listEstudiantes; let i = index ">

        <td>{{ estudiante.estado }}</td>
      </tr>
    </tbody>
  </table>
  <h3 *ngIf="listEstudiantes.length === 0">No hay registros para mostrar</h3>

</div>

import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { RouterOutlet } from '@angular/router';

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [CommonModule, RouterOutlet],
  templateUrl: './app.component.html',
  styleUrl: './app.component.css'
})
export class AppComponent {
  listEstudiantes: any[] = [
    { nombre: "Tomas Gonzalez", estado: "Promocionado" },
    { nombre: "Lucas Perez", estado: "Regular" },
    { nombre: "Patricia Funez", estado: "Libre" },

    { nombre: "Marta Perez", estado: "Promocionado" },
    { nombre: "Nicolas Gomez", estado: "Libre" },

  ]

  }
}
